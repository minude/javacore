        # тЪ║ТюгТдѓт┐х

        • тї║тѕєтљїТГЦТѕќт╝ѓТГЦ№╝ѕsynchronous/asynchronous№╝Ѕсђѓу«ђтЇЋТЮЦУ»┤№╝їтљїТГЦТў»СИђуДЇтЈ»жЮауџёТюЅт║ЈУ┐љУАїТю║тѕХ№╝їтйЊТѕЉС╗гУ┐ЏУАїтљїТГЦТЊЇСйюТЌХ№╝їтљју╗ГуџёС╗╗тіАТў»уГЅтЙЁтйЊтЅЇУ░ЃућеУ┐ћтЏъ№╝їТЅЇС╝џУ┐ЏУАїСИІСИђТГЦ№╝ЏУђїт╝ѓТГЦтѕЎуЏИтЈЇ№╝їтЁХС╗ќС╗╗тіАСИЇжюђУдЂуГЅтЙЁтйЊтЅЇУ░ЃућеУ┐ћтЏъ№╝їжђџтИИСЙЮжЮаС║ІС╗ХсђЂтЏъУ░ЃуГЅТю║тѕХТЮЦт«ъуј░С╗╗тіАжЌ┤ТгАт║ЈтЁ│у│╗сђѓ
        • тї║тѕєжў╗тАъСИјжЮъжў╗тАъ№╝ѕblocking/non-blocking№╝ЅсђѓтюеУ┐ЏУАїжў╗тАъТЊЇСйюТЌХ№╝їтйЊтЅЇу║┐уеІС╝џтцёС║јжў╗тАъуіХТђЂ№╝їТЌаТ│ЋС╗јС║ІтЁХС╗ќС╗╗тіА№╝їтЈфТюЅтйЊТЮАС╗Хт░▒у╗фТЅЇУЃйу╗Ду╗Г№╝їТ»ћтдѓ ServerSocket Тќ░У┐ъТјЦт╗║уФІт«їТ»Ћ№╝їТѕќТЋ░ТЇ«У»╗тЈќсђЂтєЎтЁЦТЊЇСйют«їТѕљ№╝ЏУђїжЮъжў╗тАътѕЎТў»СИЇу«А IO ТЊЇСйюТў»тљду╗ЊТЮЪ№╝їуЏ┤ТјЦУ┐ћтЏъ№╝їуЏИт║ћТЊЇСйютюетљјтЈ░у╗Ду╗Гтцёуљєсђѓ

        СИЇУЃйСИђТдѓУђїУ«║У«цСИ║тљїТГЦТѕќжў╗тАът░▒Тў»СйјТЋѕ№╝їтЁиСйЊУ┐ўУдЂуюІт║ћућетњїу│╗у╗ЪуЅ╣тЙЂсђѓ

        # BIO

        BIO№╝ѕblocking IO№╝Ѕ тЇ│жў╗тАъ IOсђѓТїЄуџёСИ╗УдЂТў»С╝ау╗Ъуџё java.io тїЁ№╝їт«ЃтЪ║С║јТхЂТеАтъІт«ъуј░сђѓ

        java.io тїЁТЈљСЙЏС║єТѕЉС╗гТюђуєЪуЪЦуџёСИђС║Џ IO тіЪУЃй№╝їТ»ћтдѓ File ТійУ▒АсђЂУЙЊтЁЦУЙЊтЄ║ТхЂуГЅсђѓС║цС║њТќ╣т╝ЈТў»тљїТГЦсђЂжў╗тАъуџёТќ╣т╝Ј№╝їС╣Ът░▒Тў»У»┤№╝їтюеУ»╗тЈќУЙЊтЁЦТхЂТѕќУђЁтєЎтЁЦУЙЊтЄ║ТхЂТЌХ№╝їтюеУ»╗сђЂтєЎтіеСйют«їТѕљС╣ІтЅЇ№╝їу║┐уеІС╝џСИђуЏ┤жў╗тАътюежѓБжЄї№╝їт«ЃС╗гС╣ІжЌ┤уџёУ░ЃућеТў»тЈ»жЮауџёу║┐ТђДжА║т║Јсђѓ

        java.io тїЁуџётЦйтцёТў»С╗БуаЂТ»ћУЙЃу«ђтЇЋсђЂуЏ┤УДѓ№╝Џу╝║уѓ╣тѕЎТў» IO ТЋѕујЄтњїТЅЕт▒ЋТђДтГўтюет▒ђжЎљТђД№╝їт«╣ТўЊТѕљСИ║т║ћућеТђДУЃйуџёуЊХжбѕсђѓ

        тЙѕтцџТЌХтђЎ№╝їС║║С╗гС╣ЪТіі java.net СИІжЮбТЈљСЙЏуџёжЃетѕєуйЉу╗ю API№╝їТ»ћтдѓ SocketсђЂServerSocketсђЂHttpURLConnection С╣Ътйњу▒╗тѕ░тљїТГЦжў╗тАъ IO у▒╗т║Њ№╝їтЏаСИ║уйЉу╗южђџС┐АтљїТаиТў» IO УАїСИ║сђѓ

        # NIO

        NIO№╝ѕnon-blocking IO№╝Ѕ тЇ│жЮъжў╗тАъ IOсђѓТїЄуџёТў» Java 1.4 СИГт╝ЋтЁЦуџё java.nio тїЁсђѓ

        java.nio тїЁТЈљСЙЏС║є ChannelсђЂSelectorсђЂBuffer уГЅТќ░уџёТійУ▒А№╝їтЈ»С╗ЦТъёт╗║тцџУи»тцЇућеуџёсђЂтљїТГЦжЮъжў╗тАъ IO уеІт║Ј№╝їтљїТЌХТЈљСЙЏС║єТЏ┤ТјЦУ┐ЉТЊЇСйюу│╗у╗Ът║Ћт▒ѓуџёжФўТђДУЃйТЋ░ТЇ«ТЊЇСйюТќ╣т╝Јсђѓ

        # AIO

        AIO№╝ѕAsynchronous IO№╝Ѕ тЇ│т╝ѓТГЦжЮъжў╗тАъ IO№╝їТїЄуџёТў» Java 7 СИГ№╝їт»╣ NIO ТюЅС║єУ┐ЏСИђТГЦуџёТћ╣У┐Џ№╝їС╣ЪуД░СИ║ NIO2№╝їт╝ЋтЁЦС║єт╝ѓТГЦжЮъжў╗тАъ IO Тќ╣т╝Јсђѓ

        тюе Java 7 СИГ№╝їNIO ТюЅС║єУ┐ЏСИђТГЦуџёТћ╣У┐Џ№╝їС╣Ът░▒Тў» NIO 2№╝їт╝ЋтЁЦС║єт╝ѓТГЦжЮъжў╗тАъ IO Тќ╣т╝Ј№╝їС╣ЪТюЅтЙѕтцџС║║тЈФт«Ѓ AIO№╝ѕAsynchronous IO№╝Ѕсђѓт╝ѓТГЦ IO ТЊЇСйютЪ║С║јС║ІС╗ХтњїтЏъУ░ЃТю║тѕХ№╝їтЈ»С╗Цу«ђтЇЋуљєУДБСИ║№╝їт║ћућеТЊЇСйюуЏ┤ТјЦУ┐ћтЏъ№╝їУђїСИЇС╝џжў╗тАътюежѓБжЄї№╝їтйЊтљјтЈ░тцёуљєт«їТѕљ№╝їТЊЇСйюу│╗у╗ЪС╝џжђџуЪЦуЏИт║ћу║┐уеІУ┐ЏУАїтљју╗ГтиЦСйюсђѓ

        # С║їсђЂIO ТхЂ

        ТхЂС╗јТдѓт┐хСИіТЮЦУ»┤Тў»СИђСИфУ┐ъу╗ГуџёТЋ░ТЇ«ТхЂсђѓтйЊуеІт║ЈжюђУдЂУ»╗ТЋ░ТЇ«уџёТЌХтђЎт░▒жюђУдЂСй┐ућеУЙЊтЁЦТхЂУ»╗тЈќТЋ░ТЇ«№╝їтйЊжюђУдЂтЙђтцќтєЎТЋ░ТЇ«уџёТЌХтђЎт░▒жюђУдЂУЙЊтЄ║ТхЂсђѓ

        BIO СИГТЊЇСйюуџёТхЂСИ╗УдЂТюЅСИцтцДу▒╗№╝їтГЌУіѓТхЂтњїтГЌугдТхЂ№╝їСИцу▒╗Та╣ТЇ«ТхЂуџёТќ╣тљЉжЃйтЈ»С╗ЦтѕєСИ║УЙЊтЁЦТхЂтњїУЙЊтЄ║ТхЂсђѓ

        • тГЌУіѓТхЂ

          • УЙЊтЄ║тГЌУіѓТхЂ№╝џOutputStream
        • тГЌугдТхЂ

          • УЙЊтЄ║тГЌугдТхЂ№╝џWriter

        # тГЌУіѓТхЂ

        тГЌУіѓТхЂСИ╗УдЂТЊЇСйютГЌУіѓТЋ░ТЇ«ТѕќС║їУ┐ЏтѕХт»╣У▒Асђѓ

        тГЌУіѓТхЂТюЅСИцСИфТаИт┐ЃТійУ▒Ау▒╗№╝џInputStream тњї OutputStreamсђѓТЅђТюЅуџётГЌУіѓТхЂу▒╗жЃйу╗ДТЅ┐УЄфУ┐ЎСИцСИфТійУ▒Ау▒╗сђѓ

        # ТќЄС╗ХтГЌУіѓТхЂ

        FileOutputStream тњї FileInputStream ТЈљСЙЏС║єУ»╗тєЎтГЌУіѓтѕ░ТќЄС╗ХуџёУЃйтіЏсђѓ

        ТќЄС╗ХТхЂТЊЇСйюСИђУѕгТГЦжфц№╝џ

        1. Сй┐уће File у▒╗у╗Љт«џСИђСИфТќЄС╗Хсђѓ
        2. Тіі File т»╣У▒Ау╗Љт«џтѕ░ТхЂт»╣У▒АСИісђѓ
        3. У┐ЏУАїУ»╗ТѕќтєЎТЊЇСйюсђѓ
        4. тЁ│жЌГТхЂ

        FileOutputStream тњї FileInputStream уц║СЙІ№╝џ

        public class FileStreamDemo {

    private static final String FILEPATH = "temp.log";

    public static void main(String[] args) throws Exception {
        write(FILEPATH);
        read(FILEPATH);
    }

    public static void write(String filepath) throws IOException {
        // угг1ТГЦсђЂСй┐ућеFileу▒╗ТЅЙтѕ░СИђСИфТќЄС╗Х
        File f = new File(filepath);

        // угг2ТГЦсђЂжђџУ┐ЄтГљу▒╗т«ъСЙІтїќуѕХу▒╗т»╣У▒А
        OutputStream out = new FileOutputStream(f);
        // т«ъСЙІтїќТЌХ№╝їж╗ўУ«цСИ║УдєуЏќтјЪТќЄС╗ХтєЁт«╣Тќ╣т╝Ј№╝ЏтдѓТъюТи╗тіаtrueтЈѓТЋ░№╝їтѕЎтЈўСИ║т»╣тјЪТќЄС╗ХУ┐йтіатєЁт«╣уџёТќ╣т╝Јсђѓ
        // OutputStream out = new FileOutputStream(f, true);

        // угг3ТГЦсђЂУ┐ЏУАїтєЎТЊЇСйю
        String str = "Hello World\n";
        byte[] bytes = str.getBytes();
        out.write(bytes);

        // угг4ТГЦсђЂтЁ│жЌГУЙЊтЄ║ТхЂ
        out.close();
    }

    public static void read(String filepath) throws IOException {
        // угг1ТГЦсђЂСй┐ућеFileу▒╗ТЅЙтѕ░СИђСИфТќЄС╗Х
        File f = new File(filepath);

        // угг2ТГЦсђЂжђџУ┐ЄтГљу▒╗т«ъСЙІтїќуѕХу▒╗т»╣У▒А
        InputStream input = new FileInputStream(f);

        // угг3ТГЦсђЂУ┐ЏУАїУ»╗ТЊЇСйю
        // ТюЅСИЅуДЇУ»╗тЈќТќ╣т╝Ј№╝їСйЊС╝џтЁХти«т╝ѓ
        byte[] bytes = new byte[(int) f.length()];
        int len = input.read(bytes); // У»╗тЈќтєЁт«╣
        System.out.println("У»╗тЁЦТЋ░ТЇ«уџёжЋ┐т║д№╝џ" + len);

        // угг4ТГЦсђЂтЁ│жЌГУЙЊтЁЦТхЂ
        input.close();
        System.out.println("тєЁт«╣СИ║№╝џ\n" + new String(bytes));
    }

}

        # тєЁтГўтГЌУіѓТхЂ

        ByteArrayInputStream тњї ByteArrayOutputStream Тў»ућеТЮЦт«їТѕљтєЁтГўуџёУЙЊтЁЦтњїУЙЊтЄ║тіЪУЃйсђѓ

        тєЁтГўТЊЇСйюТхЂСИђУѕгтюеућЪТѕљСИђС║ЏСИ┤ТЌХС┐АТЂ»ТЌХТЅЇСй┐ућесђѓ тдѓТъюСИ┤ТЌХС┐АТЂ»С┐ЮтГўтюеТќЄС╗ХСИГ№╝їУ┐ўжюђУдЂтюеТюЅТЋѕТюЪУ┐ЄтљјтѕажЎцТќЄС╗Х№╝їУ┐ЎТаиТ»ћУЙЃж║╗уЃдсђѓ

        ByteArrayInputStream тњї ByteArrayOutputStream уц║СЙІ№╝џ

        public class ByteArrayStreamDemo {

    public static void main(String[] args) {
        String str = "HELLOWORLD"; // т«џС╣ЅСИђСИфтГЌугдСИ▓№╝їтЁежЃеућ▒тцДтєЎтГЌТ»Їу╗ёТѕљ
        ByteArrayInputStream bis = new ByteArrayInputStream(str.getBytes());
        ByteArrayOutputStream bos = new ByteArrayOutputStream();
        // тЄєтцЄС╗јтєЁтГўByteArrayInputStreamСИГУ»╗тЈќтєЁт«╣
        int temp = 0;
        while ((temp = bis.read()) != -1) {
            char c = (char) temp; // У»╗тЈќуџёТЋ░тГЌтЈўСИ║тГЌугд
            bos.write(Character.toLowerCase(c)); // т░єтГЌугдтЈўСИ║т░ЈтєЎ
        }
        // ТЅђТюЅуџёТЋ░ТЇ«т░▒тЁежЃежЃйтюеByteArrayOutputStreamСИГ
        String newStr = bos.toString(); // тЈќтЄ║тєЁт«╣
        try {
            bis.close();
            bos.close();
        } catch (IOException e) {
            e.printStackTrace();
        }
        System.out.println(newStr);
    }

}

        # у«АжЂЊТхЂ

        у«АжЂЊТхЂуџёСИ╗УдЂСйюућеТў»тЈ»С╗ЦУ┐ЏУАїСИцСИфу║┐уеІжЌ┤уџёжђџС┐Асђѓ

        тдѓТъюУдЂУ┐ЏУАїу«АжЂЊжђџС┐А№╝їтѕЎт┐ЁжА╗Тіі PipedOutputStream У┐ъТјЦтюе PipedInputStream СИісђѓСИ║ТГц№╝їPipedOutputStream СИГТЈљСЙЏС║є connect() Тќ╣Т│Ћсђѓ

        public class PipedStreamDemo {

    public static void main(String[] args) {
        Send s = new Send();
        Receive r = new Receive();
        try {
            s.getPos().connect(r.getPis()); // У┐ъТјЦу«АжЂЊ
        } catch (IOException e) {
            e.printStackTrace();
        }
        new Thread(s).start(); // тљ»тіеу║┐уеІ
        new Thread(r).start(); // тљ»тіеу║┐уеІ
    }

    static class Send implements Runnable {

        private PipedOutputStream pos = null;

        Send() {
            pos = new PipedOutputStream(); // т«ъСЙІтїќУЙЊтЄ║ТхЂ
        }

        @Override
        public void run() {
            String str = "Hello World!!!";
            try {
                pos.write(str.getBytes());
            } catch (IOException e) {
                e.printStackTrace();
            }
            try {
                pos.close();
            } catch (IOException e) {
                e.printStackTrace();
            }
        }

        /**
         * тЙЌтѕ░ТГцу║┐уеІуџёу«АжЂЊУЙЊтЄ║ТхЂ
         */
        PipedOutputStream getPos() {
            return pos;
        }

    }

    static class Receive implements Runnable {

        private PipedInputStream pis = null;

        Receive() {
            pis = new PipedInputStream();
        }

        @Override
        public void run() {
            byte[] b = new byte[1024];
            int len = 0;
            try {
                len = pis.read(b);
            } catch (IOException e) {
                e.printStackTrace();
            }
            try {
                pis.close();
            } catch (IOException e) {
                e.printStackTrace();
            }
            System.out.println("ТјЦТћХуџётєЁт«╣СИ║№╝џ" + new String(b, 0, len));
        }

        /**
         * тЙЌтѕ░ТГцу║┐уеІуџёу«АжЂЊУЙЊтЁЦТхЂ
         */
        PipedInputStream getPis() {
            return pis;
        }

    }

}

        # ТЋ░ТЇ«ТЊЇСйюТхЂ

        ТЋ░ТЇ«ТЊЇСйюТхЂТЈљСЙЏС║єТа╝т╝ЈтїќУ»╗тЁЦтњїУЙЊтЄ║ТЋ░ТЇ«уџёТќ╣Т│Ћ№╝їтѕєтѕФСИ║ DataInputStream тњї DataOutputStreamсђѓ

        DataInputStream тњї DataOutputStream Та╝т╝ЈтїќУ»╗тєЎТЋ░ТЇ«уц║СЙІ№╝џ

        public class DataStreamDemo {

    public static final String FILEPATH = "temp.log";

    public static void main(String[] args) throws IOException {
        write(FILEPATH);
        read(FILEPATH);
    }

    private static void write(String filepath) throws IOException {
        // 1.Сй┐уће File у▒╗у╗Љт«џСИђСИфТќЄС╗Х
        File f = new File(filepath);

        // 2.Тіі File т»╣У▒Ау╗Љт«џтѕ░ТхЂт»╣У▒АСИі
        DataOutputStream dos = new DataOutputStream(new FileOutputStream(f));

        // 3.У┐ЏУАїУ»╗ТѕќтєЎТЊЇСйю
        String[] names = { "УАгУАБ", "ТЅІтЦЌ", "тЏ┤тиЙ" };
        float[] prices = { 98.3f, 30.3f, 50.5f };
        int[] nums = { 3, 2, 1 };
        for (int i = 0; i < names.length; i++) {
            dos.writeChars(names[i]);
            dos.writeChar('\t');
            dos.writeFloat(prices[i]);
            dos.writeChar('\t');
            dos.writeInt(nums[i]);
            dos.writeChar('\n');
        }

        // 4.тЁ│жЌГТхЂ
        dos.close();
    }

    private static void read(String filepath) throws IOException {
        // 1.Сй┐уће File у▒╗у╗Љт«џСИђСИфТќЄС╗Х
        File f = new File(filepath);

        // 2.Тіі File т»╣У▒Ау╗Љт«џтѕ░ТхЂт»╣У▒АСИі
        DataInputStream dis = new DataInputStream(new FileInputStream(f));

        // 3.У┐ЏУАїУ»╗ТѕќтєЎТЊЇСйю
        String name = null; // ТјЦТћХтљЇуД░
        float price = 0.0f; // ТјЦТћХС╗иТа╝
        int num = 0; // ТјЦТћХТЋ░жЄЈ
        char[] temp = null; // ТјЦТћХтЋєтЊЂтљЇуД░
        int len = 0; // С┐ЮтГўУ»╗тЈќТЋ░ТЇ«уџёСИфТЋ░
        char c = 0; // '\u0000'
        try {
            while (true) {
                temp = new char[200]; // т╝ђУЙЪуЕ║жЌ┤
                len = 0;
                while ((c = dis.readChar()) != '\t') { // ТјЦТћХтєЁт«╣
                    temp[len] = c;
                    len++; // У»╗тЈќжЋ┐т║дтіа1
                }
                name = new String(temp, 0, len); // т░єтГЌугдТЋ░у╗ётЈўСИ║String
                price = dis.readFloat(); // У»╗тЈќС╗иТа╝
                dis.readChar(); // У»╗тЈќ\t
                num = dis.readInt(); // У»╗тЈќint
                dis.readChar(); // У»╗тЈќ\n
                System.out.printf("тљЇуД░№╝џ%s№╝ЏС╗иТа╝№╝џ%5.2f№╝ЏТЋ░жЄЈ№╝џ%d\n", name, price, num);
            }
        } catch (EOFException e) {
            System.out.println("у╗ЊТЮЪ");
        } catch (IOException e) {
            e.printStackTrace();
        }

        // 4.тЁ│жЌГТхЂ
        dis.close();
    }

}

    public static void write(String filepath) throws IOException {
        // 1.Сй┐уће File у▒╗у╗Љт«џСИђСИфТќЄС╗Х
        File f = new File(filepath);

        // 2.Тіі File т»╣У▒Ау╗Љт«џтѕ░ТхЂт»╣У▒АСИі

        // тдѓТъюСИЇтЁ│жЌГТхЂ№╝їтѕЎу╝Њтє▓тї║уџётєЁт«╣Тў»ТЌаТ│ЋУЙЊтЄ║уџё

        out.flush();

    public static char[] read(String filepath) throws IOException {
        // 1.Сй┐уће File у▒╗у╗Љт«џСИђСИфТќЄС╗Х

        // 2.Тіі File т»╣У▒Ау╗Љт«џтѕ░ТхЂт»╣У▒АСИі

            c[len] = (char) temp;
            len++;
        }
        System.out.println("ТќЄС╗ХтГЌугдТЋ░СИ║№╝џ" + len);

        // 4.тЁ│жЌГТхЂ

        return c;
    }

        # СИЅсђЂС╝ау╗Ъ BIO

        тюе Linux СИГ№╝їтйЊт║ћућеУ┐ЏуеІУ░Ѓуће recvfrom Тќ╣Т│ЋУ░ЃућеТЋ░ТЇ«уџёТЌХтђЎ№╝їтдѓТъютєЁТаИТ▓АТюЅТііТЋ░ТЇ«тЄєтцЄтЦйСИЇС╝џуФІтѕ╗У┐ћтЏъ№╝їУђїТў»С╝џу╗ЈтјєуГЅтЙЁТЋ░ТЇ«тЄєтцЄт░▒у╗ф№╝їТЋ░ТЇ«С╗јтєЁТаИтцЇтѕХтѕ░ућеТѕиуЕ║жЌ┤С╣ІтљјтєЇУ┐ћтЏъ№╝їУ┐ЎТюЪжЌ┤т║ћућеУ┐ЏуеІСИђуЏ┤жў╗тАъуЏ┤тѕ░У┐ћтЏъ№╝їТЅђС╗ЦУбФуД░СИ║жў╗тАъ IO ТеАтъІсђѓ

        BIO С╝џжў╗тАъУ┐ЏуеІ№╝їСИЇжђѓтљѕжФўт╣ХтЈЉтю║ТЎ»сђѓ

        жЄЄуће BIO уџёТюЇтіАуФ»№╝їжђџтИИућ▒СИђСИфуІгуФІуџё Acceptor у║┐уеІУ┤ЪУ┤БуЏЉтљгт«бТѕиуФ»уџёУ┐ъТјЦсђѓТѕЉС╗гСИђУѕгжђџУ┐Єтюеwhile(true) тЙфуј»СИГТюЇтіАуФ»С╝џУ░Ѓуће accept() Тќ╣Т│ЋуГЅтЙЁТјЦТћХт«бТѕиуФ»уџёУ┐ъТјЦуџёТќ╣т╝ЈуЏЉтљгУ»иТ▒ѓ№╝їУ»иТ▒ѓСИђТЌдТјЦТћХтѕ░СИђСИфУ┐ъТјЦУ»иТ▒ѓ№╝їт░▒тЈ»С╗Цт╗║уФІжђџС┐АтЦЌТјЦтГЌтюеУ┐ЎСИфжђџС┐АтЦЌТјЦтГЌСИіУ┐ЏУАїУ»╗тєЎТЊЇСйю№╝їТГцТЌХСИЇУЃйтєЇТјЦТћХтЁХС╗ќт«бТѕиуФ»У┐ъТјЦУ»иТ▒ѓ№╝їтЈфУЃйуГЅтЙЁтљїтйЊтЅЇУ┐ъТјЦуџёт«бТѕиуФ»уџёТЊЇСйюТЅДУАїт«їТѕљ№╝ї СИЇУ┐ЄтЈ»С╗ЦжђџУ┐Єтцџу║┐уеІТЮЦТћ»ТїЂтцџСИфт«бТѕиуФ»уџёУ┐ъТјЦ№╝їтдѓСИітЏЙТЅђуц║сђѓ

        тдѓТъюУдЂУ«Е BIO жђџС┐АТеАтъІ УЃйтцЪтљїТЌХтцёуљєтцџСИфт«бТѕиуФ»У»иТ▒ѓ№╝їт░▒т┐ЁжА╗Сй┐ућетцџу║┐уеІ№╝ѕСИ╗УдЂтјЪтЏаТў»socket.accept()сђЂsocket.read()сђЂsocket.write() ТХЅтЈіуџёСИЅСИфСИ╗УдЂтЄйТЋ░жЃйТў»тљїТГЦжў╗тАъуџё№╝Ѕ№╝їСйєС╝џжђаТѕљСИЇт┐ЁУдЂуџёу║┐уеІт╝ђжћђсђѓСИЇУ┐ЄтЈ»С╗ЦжђџУ┐Є у║┐уеІТ▒аТю║тѕХ Тћ╣тќё№╝їу║┐уеІТ▒аУ┐ўтЈ»С╗ЦУ«Еу║┐уеІуџётѕЏт╗║тњїтЏъТћХТѕљТюгуЏИт»╣УЙЃСйјсђѓСй┐ућеFixedThreadPool тЈ»С╗ЦТюЅТЋѕуџёТјДтѕХС║єу║┐уеІуџёТюђтцДТЋ░жЄЈ№╝їС┐ЮУ»ЂС║єу│╗у╗ЪТюЅжЎљуџёУхёТ║љуџёТјДтѕХ№╝їт«ъуј░С║є N(т«бТѕиуФ»У»иТ▒ѓТЋ░жЄЈ):M(тцёуљєт«бТѕиуФ»У»иТ▒ѓуџёу║┐уеІТЋ░жЄЈ)уџёС╝фт╝ѓТГЦ I/O ТеАтъІ№╝ѕN тЈ»С╗ЦУ┐юУ┐ютцДС║ј M№╝Ѕ№╝їСИІжЮбСИђУіѓ"С╝фт╝ѓТГЦ BIO"СИГС╝џУ»ду╗єС╗Іу╗Їтѕ░сђѓ

        тЇ│Сй┐тЈ»С╗Цућеу║┐уеІТ▒ауЋЦтЙ«С╝ўтїќ№╝їСйєТў»С╝џТХѕУђЌт«ЮУ┤хуџёу║┐уеІУхёТ║љ№╝їт╣ХСИћтюеуЎЙСИЄу║Дт╣ХтЈЉтю║ТЎ»СИІС╣ЪТњЉСИЇСйЈсђѓтдѓТъют╣ХтЈЉУ«┐жЌ«жЄЈтбътіаС╝џт»╝УЄ┤у║┐уеІТЋ░ТђЦтЅДУєеУЃђтЈ»УЃйС╝џт»╝УЄ┤у║┐уеІтаєТаѕТ║бтЄ║сђЂтѕЏт╗║Тќ░у║┐уеІтц▒У┤ЦуГЅжЌ«жбў№╝їТюђу╗ѕт»╝УЄ┤У┐ЏуеІт«ЋТю║ТѕќУђЁтЃхТГ╗№╝їСИЇУЃйт»╣тцќТЈљСЙЏТюЇтіАсђѓ

        # тЏЏсђЂС╝фт╝ѓТГЦ BIO

        СИ║С║єУДБтє│тљїТГЦжў╗тАъ I/O жЮбСИ┤уџёСИђСИфжЊЙУи»жюђУдЂСИђСИфу║┐уеІтцёуљєуџёжЌ«жбў№╝їтљјТЮЦТюЅС║║т»╣т«Ѓуџёу║┐уеІТеАтъІУ┐ЏУАїС║єС╝ўтїќСИђСИђСИђтљјуФ»жђџУ┐ЄСИђСИфу║┐уеІТ▒аТЮЦтцёуљєтцџСИфт«бТѕиуФ»уџёУ»иТ▒ѓТјЦтЁЦ№╝їтйбТѕљт«бТѕиуФ»СИфТЋ░ M№╝џу║┐уеІТ▒аТюђтцДу║┐уеІТЋ░ N уџёТ»ћСЙІтЁ│у│╗№╝їтЁХСИГ M тЈ»С╗ЦУ┐юУ┐ютцДС║ј NсђѓжђџУ┐Єу║┐уеІТ▒атЈ»С╗ЦуЂхТ┤╗тю░У░ЃжЁЇу║┐уеІУхёТ║љ№╝їУ«Йуй«у║┐уеІуџёТюђтцДтђ╝№╝їжў▓ТГбућ▒С║јТхижЄЈт╣ХтЈЉТјЦтЁЦт»╝УЄ┤у║┐уеІУђЌт░йсђѓ

        жЄЄућеу║┐уеІТ▒атњїС╗╗тіАжўЪтѕЌтЈ»С╗Цт«ъуј░СИђуДЇтЈФтЂџС╝фт╝ѓТГЦуџё I/O жђџС┐АТАєТъХ№╝їт«ЃуџёТеАтъІтЏЙтдѓСИітЏЙТЅђуц║сђѓтйЊТюЅТќ░уџёт«бТѕиуФ»ТјЦтЁЦТЌХ№╝їт░єт«бТѕиуФ»уџё Socket т░ЂУБЁТѕљСИђСИф Task№╝ѕУ»ЦС╗╗тіАт«ъуј░ java.lang.Runnable ТјЦтЈБ№╝ЅТіЋжђњтѕ░тљјуФ»уџёу║┐уеІТ▒аСИГУ┐ЏУАїтцёуљє№╝їJDK уџёу║┐уеІТ▒ау╗┤ТіцСИђСИфТХѕТЂ»жўЪтѕЌтњї N СИфТ┤╗УиЃу║┐уеІ№╝їт»╣ТХѕТЂ»жўЪтѕЌСИГуџёС╗╗тіАУ┐ЏУАїтцёуљєсђѓућ▒С║ју║┐уеІТ▒атЈ»С╗ЦУ«Йуй«ТХѕТЂ»жўЪтѕЌуџётцДт░ЈтњїТюђтцДу║┐уеІТЋ░№╝їтЏаТГц№╝їт«ЃуџёУхёТ║љтЇаућеТў»тЈ»ТјДуџё№╝їТЌаУ«║тцџт░ЉСИфт«бТѕиуФ»т╣ХтЈЉУ«┐жЌ«№╝їжЃйСИЇС╝џт»╝УЄ┤УхёТ║љуџёУђЌт░йтњїт«ЋТю║сђѓ

        С╝фт╝ѓТГЦ I/O жђџС┐АТАєТъХжЄЄућеС║єу║┐уеІТ▒ат«ъуј░№╝їтЏаТГцжЂ┐тЁЇС║єСИ║Т»ЈСИфУ»иТ▒ѓжЃйтѕЏт╗║СИђСИфуІгуФІу║┐уеІжђаТѕљуџёу║┐уеІУхёТ║љУђЌт░йжЌ«жбўсђѓСИЇУ┐ЄтЏаСИ║т«Ѓуџёт║Ћт▒ѓС╗ЇуёХТў»тљїТГЦжў╗тАъуџё BIO ТеАтъІ№╝їтЏаТГцТЌаТ│ЋС╗јТа╣ТюгСИіУДБтє│жЌ«жбўсђѓ
